62template<typename Type, typename PropType = Type>
63struct ArticyPredefinedTypeInfo : public ArticyPredefinedTypeBase
64{
65public:
66 ArticyPredefinedTypeInfo(FString TypeName, FString PropTypeName, FString DefaultValue, PropType(*Deserializer)(PROP_SETTER_PARAMS))
67 : ArticyPredefinedTypeBase(TypeName, PropTypeName, DefaultValue), Deserializer(Deserializer)
68 { }
69
70 virtual ~ArticyPredefinedTypeInfo() = default;
71
72public:
73
74 typedef PropType(*TDeserializer)(PROP_SETTER_PARAMS);
76 TDeserializer Deserializer = nullptr;
77
78public:
79
81 void SetProp(FName Property, PROP_SETTER_PARAMS) override
82 {
83 if (ensure(Deserializer))
84 Model->SetProp<PropType>(Property, Deserializer(PROP_SETTER_ARGS));
85 }
86
88 void SetArray(FName ArrayProperty, ARRAY_SETTER_PARAMS) override
89 {
90 if (!ensure(Deserializer))
91 return;
92
93 static TArray<PropType> propArray;
94
95 propArray.Reset(JsonArray.Num());
96 for (auto j : JsonArray)
97 propArray.Add(Deserializer(Model, Path, j, PackageName));
98
99 Model->SetProp(ArrayProperty, propArray);
100 }

110public:
111 ArticyObjectTypeInfo(FString TypeName, FString PropTypeName) : ArticyPredefinedTypeInfo<Type, PropType>(TypeName, PropTypeName, TEXT("nullptr"), [](PROP_SETTER_PARAMS)
112 {
113 auto val = NewObject<Type>(Model);
114 val->InitFromJson(Json);
115
116 UObject* raw = val;
117 auto prim = Cast<UArticyPrimitive>(raw);
118 if (prim)
119 Model->AddSubobject(prim);
120
121 return val;
122 }) {
123 static_assert(std::is_base_of<UArticyBaseObject, Type>::value, "ArticyObjectTypeInfo might only be used for UArticyBaseObject!");
124 }
