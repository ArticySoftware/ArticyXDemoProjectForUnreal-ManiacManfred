316template<typename Lambda>
317void UArticyFlowPlayer::ShadowedOperation(Lambda Operation) const
318{
319 if (std::is_null_pointer<Lambda>::value)
320 return;
321 if (!ensure(GetGVs()))
322 {
323 UE_LOG(LogArticyRuntime, Warning, TEXT("FlowPlayer cannot get GVs!"));
324 return;
325 }
326 if (!ensure(ShadowLevel < ShadowLevelLimit))
327 {
328 UE_LOG(LogArticyRuntime, Warning, TEXT("More than 100 nested ShadowedOperations, probably there's an infinite loop in the Flow!"));
329 return;
330 }
331
332 //push shadow state
333 ++ShadowLevel;
334
335 //notify on push
336 GetGVs()->PushState(ShadowLevel);
337 GetGVs()->PushSeen();
338 UArticyDatabase::Get(this)->PushState(ShadowLevel);
339 OnShadowOpStart.Broadcast();
340
341 //execute the operation
342 Operation();
343
344 //notify on pop
345 OnShadowOpEnd.Broadcast();
346 UArticyDatabase::Get(this)->PopState(ShadowLevel);
347 GetGVs()->PopSeen();
348 GetGVs()->PopState(ShadowLevel);
349
350 //pop shadow state
351 if (ensure(ShadowLevel > 0))
352 --ShadowLevel;
353}
