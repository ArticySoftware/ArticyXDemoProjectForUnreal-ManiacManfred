310template<typename Lambda>
311void UArticyFlowPlayer::ShadowedOperation(Lambda Operation) const
312{
313 if (std::is_null_pointer<Lambda>::value)
314 return;
315 if (!ensure(GetGVs()))
316 {
317 UE_LOG(LogArticyRuntime, Warning, TEXT("FlowPlayer cannot get GVs!"));
318 return;
319 }
320 if (!ensure(ShadowLevel < ShadowLevelLimit))
321 {
322 UE_LOG(LogArticyRuntime, Warning, TEXT("More than 100 nested ShadowedOperations, probably there's an infinite loop in the Flow!"));
323 return;
324 }
325
326 //push shadow state
327 ++ShadowLevel;
328
329 //notify on push
330 GetGVs()->PushState(ShadowLevel);
331 GetGVs()->PushSeen();
332 UArticyDatabase::Get(this)->PushState(ShadowLevel);
333 OnShadowOpStart.Broadcast();
334
335 //execute the operation
336 Operation();
337
338 //notify on pop
339 OnShadowOpEnd.Broadcast();
340 UArticyDatabase::Get(this)->PopState(ShadowLevel);
341 GetGVs()->PopSeen();
342 GetGVs()->PopState(ShadowLevel);
343
344 //pop shadow state
345 if (ensure(ShadowLevel > 0))
346 --ShadowLevel;
347}
